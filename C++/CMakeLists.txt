cmake_minimum_required(VERSION 3.10)

project(FIR_SIMD)

set(SOURCES
    src/FIRFilter.cpp
    src/main.cpp
)

add_executable(FIR_SIMD ${SOURCES})

target_include_directories(FIR_SIMD 
    PUBLIC dependencies
    PUBLIC include)

if (MSVC)
    target_compile_options(FIR_SIMD 
                PRIVATE /EHsc
                PRIVATE /arch:AVX)
else()
    target_compile_options(FIR_SIMD 
                PRIVATE -mavx)
endif()

set_property(TARGET FIR_SIMD PROPERTY CXX_STANDARD 20)
